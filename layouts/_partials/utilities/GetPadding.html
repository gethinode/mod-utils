{{- /* Define main/section padding size. Reads [*.padding] x/y and optional [*.padding.mobile] x/y for the given "section" (defaults to "main"), falling back to [main.padding] when the section itself does not define a value. Returns a dict with "x" and "y" always set, plus "mobileX"/"mobileY" only when a mobile override is configured and passes validation. */ -}} {{- $section := "main" }} {{- with index . "section" }}{{ $section = . }}{{ end -}} {{- $defaultY := 5 -}} {{- $defaultX := 5 -}} {{- $permitted := slice 0 1 2 3 4 5 }} {{- $paddingX := $defaultX -}} {{- $paddingY := $defaultY -}} {{- $mobileX := "" -}} {{- $mobileY := "" -}} {{- $padding := index site.Params $section -}} {{- if $padding.padding.x }} {{ $paddingX = $padding.padding.x }} {{- else if isset site.Params.main.padding "x" }} {{ $paddingX = site.Params.main.padding.x }} {{ end -}} {{- if $padding.padding.y }} {{ $paddingY = $padding.padding.y }} {{- else if isset site.Params.main.padding "y" }} {{ $paddingY = site.Params.main.padding.y }} {{ end -}} {{- /* default dict avoids IsSet with unsupported type "invalid" warnings when the mobile table is absent */ -}} {{- $sectionMobile := default dict $padding.padding.mobile -}} {{- $mainMobile := default dict site.Params.main.padding.mobile -}} {{- if isset $sectionMobile "x" }} {{ $mobileX = index $sectionMobile "x" }} {{- else if isset $mainMobile "x" }} {{ $mobileX = index $mainMobile "x" }} {{ end -}} {{- if isset $sectionMobile "y" }} {{ $mobileY = index $sectionMobile "y" }} {{- else if isset $mainMobile "y" }} {{ $mobileY = index $mainMobile "y" }} {{ end -}} {{- if not (in $permitted $paddingX) -}} {{- partial "utilities/LogWarn.html" (dict "partial" "utilities/GetPadding.html" "warnid" "warn-invalid-padding" "msg" "Invalid padding" "details" (printf "layout [_default/baseof.html] - Invalid value for param 'padding.x': %v (type %T)" $paddingX $paddingX) "file" page.File ) -}} {{- $paddingX = $defaultX -}} {{- end -}} {{- if not (in $permitted $paddingY) -}} {{- partial "utilities/LogWarn.html" (dict "partial" "utilities/GetPadding.html" "warnid" "warn-invalid-padding" "msg" "Invalid padding" "details" (printf "layout [_default/baseof.html] - Invalid value for param 'padding.y': %v (type %T)" $paddingY $paddingY) "file" page.File ) -}} {{- $paddingY = $defaultY -}} {{- end -}} {{- if and (ne $mobileX "") (not (in $permitted $mobileX)) -}} {{- partial "utilities/LogWarn.html" (dict "partial" "utilities/GetPadding.html" "warnid" "warn-invalid-padding" "msg" "Invalid padding" "details" (printf "layout [_default/baseof.html] - Invalid value for param 'padding.mobile.x': %v (type %T)" $mobileX $mobileX) "file" page.File ) -}} {{- $mobileX = "" -}} {{- end -}} {{- if and (ne $mobileY "") (not (in $permitted $mobileY)) -}} {{- partial "utilities/LogWarn.html" (dict "partial" "utilities/GetPadding.html" "warnid" "warn-invalid-padding" "msg" "Invalid padding" "details" (printf "layout [_default/baseof.html] - Invalid value for param 'padding.mobile.y': %v (type %T)" $mobileY $mobileY) "file" page.File ) -}} {{- $mobileY = "" -}} {{- end -}} {{- $result := dict "x" $paddingX "y" $paddingY -}} {{- if ne $mobileX "" }}{{ $result = merge $result (dict "mobileX" $mobileX) }}{{ end -}} {{- if ne $mobileY "" }}{{ $result = merge $result (dict "mobileY" $mobileY) }}{{ end -}} {{- return $result -}}
